  3 /**
  4  * @param {Object} obj FrameObject(TextFrame, Rectangle, Oval, Polygon...) 
  5  * @param {String} mode v:visibleBounds, g:geometricBounds
  6  * @returns {Array} [y1,x1,y2,x2]
  7  */
  8 function get_bound (obj, mode) {
  9   if (!mode.match(/[gv]/)) {return};
 10   var bounds = mode === "v" ? obj.visibleBounds : obj.geometricBounds;
 11   return bounds;
 12 }
 13 
 14 /**
 15  * @param {Object} obj FrameObject(TextFrame, Rectangle, Oval, Polygon...) 
 16  * @param {String} mode v:visibleBounds, g:geometricBounds
 17  * @returns {Array} [width, height] 
 18  */
 19 function get_size (obj, mode) {
 20   var tmp_bon = get_bound (obj, mode);
 21   if (tmp_bon === undefined) {return};
 22   var w = tmp_bon[3] - tmp_bon[1];
 23   var h = tmp_bon[2] - tmp_bon[0];
 24   return [w, h];
 25 }

 27 /**
 28  * @param {Object} obj FrameObject
 29  * @param {Number} value plus value to increase, minus value to decrease
 30  */
 31 function inc_top (obj, value) {
 32   var bounds = get_bound(obj, 'g');
 33   obj.geometricBounds = [ bounds[0] - value, bounds[1], bounds[2], bounds[3] ];
 34 }
 35 /**
 36  * @param {Object} obj FrameObject

 38  */
 39 function inc_left (obj, value) {
 40   var bounds = get_bound(obj, 'g');
 41   obj.geometricBounds = [ bounds[0], bounds[1] - value, bounds[2], bounds[3] ];
 42 }
 43 /**
 44  * @param {Object} obj FrameObject

 46  */
 47 function inc_bottom (obj, value) {
 48   var bounds = get_bound(obj, 'g');
 49   obj.geometricBounds = [ bounds[0], bounds[1], bounds[2] + value, bounds[3] ];
 50 }
 51 /**
 52  * @param {Object} obj FrameObject
 53  * @param {Number} value plus value to increase, minus value to decrease
 54  */
 55 function inc_right (obj, value) {
 56   var bounds = get_bound(obj, 'g');
 57   obj.geometricBounds = [ bounds[0], bounds[1], bounds[2], bounds[3] + value ];
 58 }
