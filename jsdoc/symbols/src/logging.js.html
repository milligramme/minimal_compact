  3 /**
  4  * @param {String} file_path  
  5  * @returns {Array} Return array of source text
  6  */
  7 function import_source (file_path) {
  8   var file_obj = new File(file_path);
  9   file_obj.encoding = "UTF-8";
 10   file_obj.open('r');
 11   var ar = [];
 12   if (file_obj) {
 13     while(!file_obj.eof){
 14       ar.push(file_obj.readln());
 15     }
 16   };
 17   file_obj.close();
 18   return ar
 19 }

 25 function export_source (file_path, content, mode) {
 26   if (mode === undefined) {mode = 'w'};
 27   if (mode === 'w' || mode === 'e') {
 28     var file_obj = new File(file_path);
 29     file_obj.encoding = "UTF-8";
 30     file_obj.open(mode);
 31     if (mode === 'e') {
 32       file_obj.seek(0,2);
 33     };
 34     if (file_obj) {
 35       file_obj.write(content);
 36     };
 37     file_obj.close();
 38   };
