  3 /**
  4  * @param {Boolean} addnew If true when no document create document
  5  * @param {Boolean} visible Default is true, if false document create hidden mode
  6  * @returns {Object} document
  7  */
  8 function def_doc (addnew, visible) {
  9   var doc;
 10   if (visible === undefined) {visible = true;};
 11   if (!addnew) {
 12     doc = app.documents[0];
 13   }
 14   else{
 15     doc = app.documents.length === 0 ? app.documents.add(visible): app.documents[0];
 16   }
 17   return doc;
 18 }

 20 /**
 21  * @param {Object} doc Document
 22  * @returns {Object} [width, height]
 23  * @example
 24  * $.writeln(get_doc_info(doc_obj).toSource());
 25  * $.writeln(get_doc_info(doc_obj)['size']);
 26  */
 27 function get_doc_info (doc) {
 28   var doc_pref = doc.documentPreferences;
 29   var doc_info = {};
 30   doc_info['size'] = [doc_pref.pageWidth, doc_pref.pageHeight];
 31   doc_info['facing'] = doc_pref.facingPages;
 32   doc_info['colomn_dir'] = doc_pref.columnDirection === HorizontalOrVertical.HORIZONTAL ? "H" : "V"; 
 33   doc_info['pages'] = doc_pref.pagesPerDocument;
 34   switch(doc_pref.pageBinding){
 35     case 1819570786: /*PageBindingOptions.LEFT_TO_RIGHT*/ doc_info['binding'] = "L";break;
 36     case 1920232546: /*PageBindingOptions.RIGHT_TO_LEFT*/doc_info['binding'] = "R";break;
 37     default: /*PageBindingOptions.DEFAULT_VALUE*/doc_info['binding'] = "D"; break;
 38   }  
 39   return doc_info;
 40 }
