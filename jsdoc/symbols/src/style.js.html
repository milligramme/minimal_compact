  3 /**
  4  * @param {Object} target Parent Object App, Document
  5  * @param {String} style_type characterStyles, paragraphStyles,,.,
  6  * @param {String} name Name to find
  7  * @returns {Array} [True, parent] if exist, [false, null] if not
  8  * @example
  9  * var result = [
 10  *   check_style_exist (app.documents[0],'characterStyles','bold'),
 11  *   check_style_exist (app.documents[0],'paragraphStyles','base'),
 12  *   check_style_exist (app.documents[0],'objectStyles','anchor'),
 13  *   check_style_exist (app.documents[0],'tableStyles','main'),
 14  *   check_style_exist (app.documents[0],'cellStyles','head')
 15  * ];
 16  * $.writeln(result);
 17  *
 18  */
 19 function check_style_exist (target, style_type, name) {
 20   // if not belongs to style group
 21   var exist;
 22   exist = (target[style_type].itemByName(name) !== null);
 23   
 24   // if not exist in root, search in group
 25   if ( !exist ) {
 26     var style_group = style_type.replace(/Styles/,"StyleGroups");
 27     if(target[style_group].length === 0) return;
 28     var style_ingroup = target[style_group].everyItem()[style_type].itemByName(name);
 29     if (style_ingroup !== null) {
 30       return [true, style_ingroup.parent.name];
 31     }
 32     else {
 33       return [false, null];
 34     }
 35   };
 36   
 37   return [exist, "root"];
 38 }
